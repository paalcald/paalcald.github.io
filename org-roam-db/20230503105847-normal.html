Estimador de Máxima Verosimilitud

\begin{equation} \mu_{\text{EMV}} = \frac{\sum_{i = 1}^n x_i}{n} = \overline{x} \end{equation} \begin{equation} \sigma_{\text{EMV}} = \sqrt{\frac{\sum_{i = 1}^n \left( x_i - \overline{x} \right)^2}{n}}= s_{n} \end{equation}

¿Por qué?   proof demostración

\begin{equation*} L(\mu, \sigma ;x) = \frac{1}{\sigma \sqrt{2\pi}} e ^{- \frac{1}{2} \left( \frac{x - \mu}{\sigma} \right)^2} \end{equation*} \begin{equation*} L(\mu, \sigma ;\vec{x}) = {\left( \frac{1}{\sigma \sqrt{2\pi}} \right)}^n e ^{- \frac{1}{2} \sum {\left( \frac{x_i - \mu}{\sigma} \right)}^2} \end{equation*} \begin{equation*} \ln L(\mu, \sigma ;\vec{x})= - n \ln \sigma - \frac{n}{2} \ln 2 \pi - \frac{1}{2 \sigma^2} \sum \left( x_i - \mu \right)^2 \end{equation*} \begin{equation*} \frac{\partial \ln L(\mu, \sigma ;\vec{x})}{\partial \mu} = \frac{1}{2 \sigma^2} \sum \left( x_i - \mu \right) \end{equation*} \begin{equation*} \frac{\partial \ln L(\mu, \sigma ;\vec{x})}{\partial \mu} = 0 \\ \iff \frac{1}{2 \sigma^2} \sum \left( x_i - \mu \right) = 0 \\ \iff \mu = \frac{\sum x_i}{n} = \overline{X} \\ \end{equation*}

Observese que

\begin{align*} \ln L(\mu, \sigma ;\vec{x}) &= - n \ln \sigma - \frac{n}{2} \ln 2 \pi - \frac{1}{2 \sigma^2} \sum \left( (x_i - \overline{x}) + (\overline{x} - \mu )\right)^2 \\ &= - n \ln \sigma - \frac{n}{2} \ln 2 \pi - \frac{1}{2 \sigma^2} \frac{\sum (x_i - \overline{x})^2 + \sum (\overline{x} - \mu_0 )^2 + (\overline{x} - \mu_0)\cancelto{0}{\sum (x_i - \overline{x})}}{n}\\ \end{align*}

Por lo tanto para encontrar el máximo para \(\sigma\) podemos fijarnos en

\begin{equation*} \ln L(\overline{x}, \sigma ;\vec{x})- n \ln \sigma - \frac{n}{2} \ln 2 \pi - \frac{1}{2 \sigma^2} \sum \left(x_i - \overline{x}\right)^2 \\ \end{equation*} \begin{equation*} \frac{\partial \ln L(\overline{x}, \sigma ;\vec{x})}{\partial \sigma} = - \frac{n}{\sigma} + \frac{1}{\sigma^3} \sum \left( x_i - \overline{x}\right)^2 \end{equation*} \begin{equation*} \frac{\partial \ln L(\overline{x}, \sigma ;\vec{x})}{\partial \sigma} = 0 \iff \frac{n}{\sigma} = \frac{1}{\sigma^3} \sum \left( x_i - \overline{x} \right)^2 \iff \sigma^2 = \frac{\sum \left( x_i - \overline{x} \right)^2}{n} \end{equation*}

Concluimos

\begin{equation} \mu_{\text{EMV}} = \frac{\sum_{i = 1}^n x_i}{n} = \overline{x} \end{equation} \begin{equation} \sigma_{\text{EMV}} = \sqrt{\frac{\sum_{i = 1}^n \left( x_i - \overline{x} \right)^2}{n}}= s_{n} \end{equation}
