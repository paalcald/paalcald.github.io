Obtener razonadamente la región crítica de los siguientes tests

\( X \sim N(\mu, \sigma) \) con \(\sigma^2\) desconocida cuando se toma una m.a.s. de esta población y se desea constrastar \( H_0 : \mu = \mu_0 \) vs \( H_1 : \mu \ne \mu_0 \).

Para realizar este tipo de hipótesis es posible realizar un test de razón de verosimilitudes.

\begin{equation*} W_n = \left\{ \vec{x} \left| \frac{\sup_{\mu, \sigma \in H} L(\mu, \sigma ;\vec{x}) }{\sup_{\mu,\sigma \in H_0} L(\mu, \sigma ;\vec{x})}\right. \ge k \right\} \end{equation*}

Antes de continuar hemos de identificar el numerador y denominador.

  1. \( \sup_{\mu, \sigma \in H} L(\mu, \sigma ;\vec{x}) \) es por definición del EMV \( L(\mu_{\text{EMV}}, \sigma_{\text{EMV}}; \vec{x}) \)
  2. \( \sup_{\mu, \sigma \in H_0} L(\mu, \sigma ;\vec{x}) \) es encontrar el valor que maximiza la verosimilitud dado \( \mu = \mu_0 \)

Encontrar el EMV para \( \mu \) y \(\sigma\).

\begin{equation*} L(\mu, \sigma ;x) = \frac{1}{\sigma \sqrt{2\pi}} e ^{- \frac{1}{2} \left( \frac{x - \mu}{\sigma} \right)^2} \end{equation*} \begin{equation*} L(\mu, \sigma ;\vec{x}) = {\left( \frac{1}{\sigma \sqrt{2\pi}} \right)}^n e ^{- \frac{1}{2} \sum {\left( \frac{x_i - \mu}{\sigma} \right)}^2} \end{equation*} \begin{equation*} \ln L(\mu, \sigma ;\vec{x})= - n \ln \sigma - \frac{n}{2} \ln 2 \pi - \frac{1}{2 \sigma^2} \sum \left( x_i - \mu \right)^2 \end{equation*} \begin{equation*} \frac{\partial \ln L(\mu, \sigma ;\vec{x})}{\partial \mu} = \frac{1}{2 \sigma^2} \sum \left( x_i - \mu \right) \end{equation*} \begin{equation*} \frac{\partial \ln L(\mu, \sigma ;\vec{x})}{\partial \mu} = 0 \\ \iff \frac{1}{2 \sigma^2} \sum \left( x_i - \mu \right) = 0 \\ \iff \mu = \frac{\sum x_i}{n} = \overline{X} \\ \end{equation*}

Observese que

\begin{align*} \ln L(\mu, \sigma ;\vec{x}) &= - n \ln \sigma - \frac{n}{2} \ln 2 \pi - \frac{1}{2 \sigma^2} \sum \left( (x_i - \overline{x}) + (\overline{x} - \mu )\right)^2 \\ &= - n \ln \sigma - \frac{n}{2} \ln 2 \pi - \frac{1}{2 \sigma^2} \frac{\sum (x_i - \overline{x})^2 + \sum (\overline{x} - \mu_0 )^2 + (\overline{x} - \mu_0)\cancelto{0}{\sum (x_i - \overline{x})}}{n}\\ \end{align*}

Por lo tanto para encontrar el máximo para \(\sigma\) podemos fijarnos en

\begin{equation*} \ln L(\overline{x}, \sigma ;\vec{x})- n \ln \sigma - \frac{n}{2} \ln 2 \pi - \frac{1}{2 \sigma^2} \sum \left(x_i - \overline{x}\right)^2 \\ \end{equation*} \begin{equation*} \frac{\partial \ln L(\overline{x}, \sigma ;\vec{x})}{\partial \sigma} = - \frac{n}{\sigma} + \frac{1}{\sigma^3} \sum \left( x_i - \overline{x}\right)^2 \end{equation*} \begin{equation*} \frac{\partial \ln L(\overline{x}, \sigma ;\vec{x})}{\partial \sigma} = 0 \iff \frac{n}{\sigma} = \frac{1}{\sigma^3} \sum \left( x_i - \overline{x} \right)^2 \iff \sigma^2 = \frac{\sum \left( x_i - \overline{x} \right)^2}{n} \end{equation*}

Concluimos

\begin{equation} \label{eq:1} \mu_{\text{EMV}} = \frac{\sum_{i = 1}^n x_i}{n} = \overline{x} \end{equation} \begin{equation} \label{eq:2} \sigma_{\text{EMV}} = \sqrt{\frac{\sum_{i = 1}^n \left( x_i - \overline{x} \right)^2}{n}}= s_{n} \end{equation}

Encontrar \(\sigma_0 \) , el valor de \( \sigma \) tal que \(L(\mu_0, \sigma ;\vec{x}) \) alcanza su máximo.

\begin{equation*} \frac{\partial \ln L(\mu_0, \sigma ;\vec{x})}{\partial \sigma} = 0 \iff \frac{n}{\sigma} = \frac{1}{\sigma^3} \sum \left( x_i - \mu_0 \right)^2 \iff \sigma^2 = \frac{\sum \left( x_i - \mu_0 \right)^2}{n} \end{equation*}

Denotemoslo por \( \)\(\sigma_0\)

\begin{equation} \label{eq:3} \sigma_0 = \sqrt{\frac{\sum \left( x_i - \mu_0 \right)^2 }{n}} \end{equation}

Lo obtendremos su relación con \( s_n \) puesto que nos será util más adelante.

\begin{align*} \sigma^2_0 &= \frac{\sum \left( x_i - \mu_0 \right)^2 }{n} \\ &= \frac{\sum \left( (x_i - \overline{x}) + (\overline{x} - \mu_0 )\right)^2 }{n} && \text{sumo y sustraigo una constante}\\ &= \frac{\sum (x_i - \overline{x})^2 + \sum (\overline{x} - \mu_0 )^2 + (\overline{x} - \mu_0)\cancelto{0}{\sum (x_i - \overline{x})}}{n} && \text{desarrollo el cuadrado}\\ &= s^2_n + (\overline{x} - \mu_0)^2 \end{align*} \begin{equation} \label{eq:4} \sigma^2_0 = s_n^2 + (\overline{x} - \mu_0)^2 \end{equation}

Calcular el coeficiente.

De \ref{eq:1} \ref{eq:2} y \ref{eq:3} obtenemos que

\begin{align*} W_n &= \left\{ \vec{x} \left| \frac{\sup_{\mu, \sigma \in H} L(\mu, \sigma ;\vec{x}) }{\sup_{\mu,\sigma \in H_0} L(\mu, \sigma ;\vec{x})}\right. \ge k \right\} \\ &= \left\{ \vec{x} \left| \frac{{\left( \frac{1}{\sigma_{\text{EMV}} \sqrt{2\pi}} \right)}^n e ^{- \frac{1}{2} \sum {\left( \frac{x_i - \mu_{\text{EMV}}}{\sigma_{\text{EMV}}} \right)}^2} } { {\left( \frac{1}{\sigma_0 \sqrt{2\pi}} \right)}^n e ^{- \frac{1}{2} \sum {\left( \frac{x_i - \mu_0}{\sigma_0} \right)}^2}}\right. \ge k \right\} \\ &= \left\{ \vec{x} \left| \frac{{\left( \frac{1}{s_n \sqrt{2\pi}} \right)}^n e ^{- \frac{1}{2} \sum {\left( \frac{x_i - \overline{x}}{s_n} \right)}^2} } { {\left( \frac{1}{\sigma_0 \sqrt{2\pi}} \right)}^n e ^{- \frac{1}{2} \sum {\left( \frac{x_i - \mu_0}{\sigma_0} \right)}^2}}\right. \ge k \right\} \\ &= \left\{ \vec{x} \left| \frac{{\left( \frac{1}{s_n \sqrt{2\pi}} \right)}^n e ^{- \frac{n}{2}}} { {\left( \frac{1}{\sigma_0 \sqrt{2\pi}} \right)}^n e ^{- \frac{n}{2} }}\right. \ge k \right\} \\ &= \left\{ \vec{x} \left| \left( \frac{\sigma_0} {s_n} \right)^n\right. \ge k \right\} \\ &= \left\{ \vec{x} \left| \left( \frac{\sigma^2_0} {s^2_n} \right)^{n/2}\right. \ge k \right\} \\ &= \left\{ \vec{x} \left| \left( \frac{s^2_n + (\overline{x} - \mu_0)^2} {s^2_n} \right)^{n/2}\right. \ge k \right\} && \text{ donde hemos usado \ref{eq:4}}\\ &= \left\{ \vec{x} \left| 1 + \frac{(\overline{x} - \mu_0)^2} {s_n^2} \right. \ge k' \right\} && \text{por ser la raíz monótona} \\ &= \left\{ \vec{x} \left| \frac{(\overline{x} - \mu_0)^2} {s_n^2} \right. \ge k'' \right\}\\ &= \left\{ \vec{x} \left| \frac{n(\overline{x} - \mu_0)^2} {s^2} \right. \ge k''' \right\} && (n)s^2_n = (n-1)s^{2} \end{align*} \begin{equation} \label{eq:5} \left. \begin{aligned} X \sim N(\mu, \sigma) \\ \sigma \text{ desconocida} \end{aligned} \right\} \implies Z = \frac{\sqrt{n}(X-\mu)}{s} \sim t_{n-1} \end{equation}

Por lo que el Teorema de Neyman-Pearson nos permite concluir que

\begin{equation*} \alpha = P(\mu_0 \in W_n) = P(|k \le Z|) \implies k = t_{n-1;\alpha/2} \end{equation*}

De donde obtenemos finalmente la región crítica de nuestro test \( H_0 : \mu = \mu_0 \) vs \( H_1 : \mu \ne \mu_0 \) con certeza \(1 - \alpha\) como \(\left( \overline{x} - \frac{s}{\sqrt{n}} \cdot t_{n-1;\alpha/2}, \overline{x} + \frac{s}{\sqrt{n}} \cdot t_{n-1;\alpha/2} \right) \)

ó alternativamente

\(\left( \overline{x} - \frac{s_n}{\sqrt{n - 1}} \cdot t_{n-1;\alpha/2}, \overline{x} + \frac{s_n}{\sqrt{n - 1}} \cdot t_{n-1;\alpha/2} \right) \)

Aceptaremos la hipótesis nula siempre y cuando \( \mu_0 \) pertenezca a este intervalo una vez introduzcamos los valores muestrales y el t-valor.

\( X \sim N(\mu_1,\sigma) \), \( Y \sim N(\mu_2, \sigma) \) con \( \sigma^2 \) desconocida. Se toman dos m.a.s independientes de tamaños n y m i.e. \( X = (x_1, ..., x_n) \) e \(Y = (y_1, ..., y_m) \). Contrastar \( H_0: \mu_1 = \mu_2 \) vs \( H_1: \mu_1 \ne \mu_2 \).

Como en el ejercicio anterior usaremos un test de razón de verosimilitudes.

\begin{equation*} W_n = \left\{ \vec{x} \left| \frac{\sup_{\mu_1, \mu_2, \sigma \in H} L(\mu_1, \mu_2, \sigma ;\vec{x}, \vec{y}) }{\sup_{\mu_{1}, \mu_2,\sigma \in H_0} L(\mu_1, \mu_2,\sigma ;\vec{x}, \vec{y})}\right. \ge k \right\} \end{equation*} \begin{equation*} L(\mu_1, \mu_2,\sigma ;\vec{x}, \vec{y}) = \left( \frac{1}{\sigma \sqrt{2\pi}} \right)^{n + m} e^{ -\frac{1}{2\sigma^2} \left( \sum_{i=1}^n\left(x_i - \mu_1\right)^2 + \sum_{i=1}^m\left(y_i - \mu_2\right)^2 \right) } \end{equation*} \begin{equation*} \ln L(\mu_1, \mu_2,\sigma ;\vec{x}, \vec{y}) = - (n + m) \ln \sigma - \frac{(n + m)}{2} \ln 2\pi -\frac{1}{2\sigma^2} \left( \sum_{i=1}^n\left(x_i - \mu_1\right)^2 + \sum_{i=1}^m\left(y_i - \mu_2\right)^2 \right) \end{equation*}

Donde podemos usar el mismo truco que en el apartado anterior para sólo preocuparnos de una parcial cada vez para obtener los EMV

\begin{equation} \label{eq:6} {\mu_1}_{\text{EMV}} = \overline{x} \end{equation} \begin{equation} \label{eq:7} {\mu_2}_{\text{EMV}} = \overline{y} \end{equation} \begin{equation} \label{eq:8} {\sigma^2}_{\text{EMV}} = \frac{\sum_{i=1}^n\left(x_i - \overline{x}\right)^2 + \sum_{i=1}^m\left(y_i - \overline{y}\right)^2}{n + m} = \frac{(n-1) s^2_x + (m - 1) s^2_y}{n + m} \end{equation}

Y en el caso de la hipótesis \( \mu_1 = \mu_2 \) tenemos

\begin{equation*} \ln L(\mu,\sigma ;\vec{x}, \vec{y}) = - (n + m) \ln \sigma - \frac{(n + m)}{2} \ln 2\pi -\frac{1}{2\sigma^2} \left( \sum_{i=1}^n\left(x_i - \mu\right)^2 + \sum_{i=1}^m\left(y_i - \mu\right)^2 \right) \end{equation*}

Que se maximiza para un \( \mu \) que denominaremos \(\mu_{12} = \frac{n \overline{x} + m \overline{y}}{n + m}\)

ahora análogamente al apartado anterior

\begin{align*} \sum_{i=1}^n\left(x_i - \mu\right)^2 + \sum_{i=1}^m\left(y_i - \mu\right)^2 &= \sum_{i=1}^n\left((x_i - \mu_{12}) + (\mu_{12} - \mu\right))^2 + \sum_{i=1}^m\left((y_i - \mu_{12}) + (\mu_{12} - \mu\right))^2 \\ &= \sum_{i=1}^n(x_i - \mu_{12})^2 + \sum_{i=1}^n(\mu_{12} - \mu)^2 + (\mu_{12} - \mu) \sum_{i=1}^n (x_i - \mu_{12}) + \sum_{i=1}^m(y_i - \mu_{12})^2 + \sum_{i=1}^m(\mu_{12} - \mu)^2 + (\mu_{12} - \mu) \sum_{i=1}^m (y_i - \mu_{12})\\ &= \sum_{i=1}^n(x_i - \mu_{12})^2 + \sum_{i=1}^m(y_i - \mu_{12})^2 + \sum_{i=1}^{n+m}(\mu_{12} - \mu)^2 + (\mu_{12} - \mu) \left( n (x_i - \mu_{12}) + m (y_i - \mu_{12}) \right)\\ &= \sum_{i=1}^n(x_i - \mu_{12})^2 + \sum_{i=1}^m(y_i - \mu_{12})^2 + \sum_{i=1}^{n+m}(\mu_{12} - \mu)^2 + (\mu_{12} - \mu) \left( n (\frac{\cancel{n\overline{x}} + m \overline{x} - \cancel{n\overline{x}} - m\overline{y}}{n + m}) + m (\frac{n\overline{y} + \cancel{m \overline{y}} - n\overline{x} - \cancel{m\overline{y}}}{n + m}) \right)\\ &= \sum_{i=1}^n(x_i - \mu_{12})^2 + \sum_{i=1}^m(y_i - \mu_{12})^2 + \sum_{i=1}^{n+m}(\mu_{12} - \mu)^2 + \cancelto{0}{(\mu_{12} - \mu) \left( \frac{nm (\overline{x} - \overline{y})}{n + m} + \frac{nm (\overline{y} - \overline{x})}{n + m} \right)}\\ \end{align*}

Que puedo usar para encontrar el \(\sigma\) que denominaremos \(\sigma_{12}\) tal que \(L(\mu_0, \sigma ;\vec{x}, \vec{y}) \) alcanza su máximo

\begin{equation} \label{eq:9} \sigma_{12}^2 = \frac{\sum_{i=1}^n\left(x_i - \mu_{12}\right)^2 + \sum_{i=1}^m\left(y_i - \mu_{12}\right)^2}{n + m} \end{equation}

aqui haremos un pequeño detour para asociarlo a \( {\sigma^2}_{\text{EMV}} \)

\begin{align*} {\sigma^2}_{12} &= \sum_{i=1}^n\left(x_i - \mu_{12}\right)^2 + \sum_{i=1}^m\left(y_i - \mu_{12}\right)^2\\ &= \sum_{i=1}^n\left((x_i - \overline{x}) + (\overline{x} - \mu_{12}\right))^2 + \sum_{i=1}^m\left((y_i - \overline{y}) + (\overline{y} - \mu_{12}\right))^2 \\ &= \sum_{i=1}^n(x_i - \overline{x})^2 + \sum_{i=1}^m(y_i - \overline{y})^2 + \sum_{i=1}^n(\overline{x} - \mu_{12})^2 + \sum_{i=1}^m(\overline{y} - \mu_{12})^2 + \cancelto{0}{(\overline{x} - \mu_{12}) \sum_{i=1}^n (x_i - \overline{x})} + \cancelto{0}{(\overline{y} - \mu_{12}) \sum_{i=1}^m (y_i - \overline{y})}\\ &= (n + m) \sigma^2_{\text{EMV}} + n(\overline{x} - \mu_{12})^2 + m(\overline{y} - \mu_{12})^2 \\ &= (n + m) \sigma^2_{\text{EMV}} + n\left( \frac{\cancel{n\overline{x}} + m\overline{x} \cancel{- n\overline{x}} - m\overline{y}}{n + m} \right)^2 + m\left( \frac{n\overline{y} + \cancel{m\overline{y}} - n\overline{x} \cancel{- m\overline{y}}}{n + m} \right)^2\\ &= (n + m) \sigma^2_{\text{EMV}} + nm\left( \frac{\overline{x} - \overline{y}}{n + m} \right)^2 + nm\left( \frac{\overline{y} - \overline{x}}{n + m} \right)^2\\ &= (n + m) \sigma^2_{\text{EMV}} + 2nm\left( \frac{\overline{x} - \overline{y}}{n + m} \right)^2\\ \end{align*}

ya nos vemos capaces de sustituir en la región crítica

\begin{align*} W_n &= \left\{ \vec{x} \left| \frac {\sup_{\mu, \sigma \in H} L(\mu, \sigma ;\vec{x},\vec{y}) } {\sup_{\mu,\sigma \in H_0} L(\mu, \sigma ;\vec{x}, \vec{y})}\right. \ge k \right\} \\ &= \left\{ \vec{x}, \vec{y} \left| \frac {{\left( \frac{1}{\sigma_{\text{EMV}} \sqrt{2\pi}} \right)}^{n+m} e ^{- \frac{\left( \sum^n (x_i - \mu_{\text{EMV}})^2 + \sum^m (y_i - \mu_{\text{EMV}})^2 \right)}{2 \sigma_{\text{EMV}}^2} }} {{\left( \frac{1}{\sigma_{12} \sqrt{2\pi}} \right)}^{n+m} e ^{- \frac{\left( \sum^n (x_i - \mu_{12})^2 + \sum^m (y_i - \mu_{12})^2 \right)}{2 \sigma_{12}^2} }}\right. \ge k \right\} \\ &= \left\{ \vec{x} , \vec{y}\left| \frac {{\left( \frac{1}{\sigma_{\text{EMV}}} \right)}^{n+m}} {{\left( \frac{1}{\sigma_{12}} \right)}^{n+m}}\right. \ge k \right\} \\ &= \left\{ \vec{x}, \vec{y} \left| \left( \frac{\sigma^2_{12}} {\sigma^2_{\text{EMV}}} \right) ^{\frac{n+m}{2}}\right. \ge k \right\} \\ &= \left\{ \vec{x} , \vec{y}\left| \frac{\sigma^2_{12}} {\sigma^2_{\text{EMV}}} \right. \ge k' \right\} \\ &= \left\{ \vec{x} , \vec{y}\left| \frac{(n + m) \sigma^2_{\text{EMV}} + 2nm\left( \frac{\overline{x} - \overline{y}}{n + m} \right)^2 } {\sigma^2_{\text{EMV}}} \right. \ge k' \right\} \\ &= \left\{ \vec{x} , \vec{y}\left| \frac{ \left( \overline{x} - \overline{y} \right)^2 } { \left( \frac{1}{n} + \frac{1}{m} \right) {\sigma^2_{\text{EMV}}}} \right. \ge k'' \right\} \\ &= \left\{ \vec{x} , \vec{y}\left| \frac{(n + m -2) \left( \overline{x} - \overline{y} \right)^2 } { \left(\frac{1}{n} + \frac{1}{m} \right) {\sigma^2_{p}}} \right. \ge k''' \right\} \\ \end{align*}

Que sabemos sigue una distribución t-student de n + m - 2 grados de libertad.

Como en el apartado anterior sólo queda comprobar que 0 esté en \( \left( \overline{X} - \overline{Y} - \sqrt{\frac{1}{n} + \frac{1}{m}} \cdot \sqrt{\frac{m S^2_{x} + nS^2_m}{m + n - 2}} \cdot t_{n + m - 2; \alpha /2}, \overline{X} - \overline{Y} + \sqrt{\frac{1}{n} + \frac{1}{m}} \cdot \sqrt{\frac{m S^2_{x} + nS^2_m}{m + n - 2}} \cdot t_{n + m - 2; \alpha /2} \right) \)
